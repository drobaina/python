import numpy as np
dados = [vendas['Produto A'], vendas['Produto B'], vendas['Produto C'], vendas['Produto D'], vendas['Produto E']]

medias = [np.mean(dataset) for dataset in dados]

fig, ax = plt.subplots()
bp = ax.boxplot(dados, patch_artist=True)

### range para inserir o ponto vermelho no gráfico
for i, media in enumerate(medias, 1):  # começar em 1 para coincidir com as posições dos produtos
    ax.plot(i, media, 'ro')
# 'ro': é uma string que define a aparência do ponto:
# 'r' significa que a cor do ponto é vermelha (red).
# 'o' significa que o ponto é representado por um círculo.
# 'o', 's', '*', 'D' (diamond)

media_geral = np.mean(vendas[produtos].values)

# Adiciona linha azul representando a média geral
ax.axhline(y= media_geral, color='blue', linestyle='--', label=f"Média Geral: {media_geral:.2f}")

ax.set_title("Boxplot das Vendas dos Produtos com Médias e Média Geral")
ax.set_xlabel("Produtos")
ax.set_ylabel("Vendas")
ax.set_xticklabels(['Produto A', 'Produto B', 'Produto C', 'Produto D', 'Produto E'])
ax.legend()

plt.tight_layout()
plt.show()
